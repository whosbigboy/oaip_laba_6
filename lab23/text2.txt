int toUpperAll(int c) {
	int new_c = c;
	if (c >= 'a' && c <= 'z') new_c = 'A' + (c - 'a');
	if (c >= 'а' && c <= 'п') new_c = 'А' + (c - 'а');
	if (c >= 'р' && c <= 'я') new_c = 'Р' + (c - 'Р');
	if (c == 'ё') new_c = 'Ё';
	return new_c;
}
